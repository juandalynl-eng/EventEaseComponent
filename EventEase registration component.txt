@page "/events/{EventId:int}/register"
@inject SessionStore SessionStore
@inject AttendanceTracker AttendanceTracker
@inject NavigationManager Nav

<EditForm Model="Registration" OnValidSubmit="HandleSubmit">
    <DataAnnotationsValidator />
    <ValidationSummary />

    <div class="mb-3">
        <InputText @bind-Value="Registration.FirstName" class="form-control" placeholder="First Name" />
        <ValidationMessage For="@(() => Registration.FirstName)" />
    </div>

    <div class="mb-3">
        <InputText @bind-Value="Registration.LastName" class="form-control" placeholder="Last Name" />
        <ValidationMessage For="@(() => Registration.LastName)" />
    </div>

    <div class="mb-3">
        <InputText @bind-Value="Registration.Email" class="form-control" placeholder="Email" />
        <ValidationMessage For="@(() => Registration.Email)" />
    </div>

    <button class="btn btn-primary">Register</button>
</EditForm>

@code {
    [Parameter] public int EventId { get; set; }

    public RegistrationModel Registration { get; set; } = new();

    protected override void OnInitialized()
    {
        if (SessionStore.CurrentUser is not null)
        {
            Registration = new RegistrationModel
            {
                FirstName = SessionStore.CurrentUser.FirstName,
                LastName = SessionStore.CurrentUser.LastName,
                Email = SessionStore.CurrentUser.Email
            };
        }

        if (AttendanceTracker.IsRegistered(EventId, Registration.Email))
        {
            Nav.NavigateTo($"/events/{EventId}/already-registered");
        }
    }

    private void HandleSubmit()
    {
        if (AttendanceTracker.IsRegistered(EventId, Registration.Email))
        {
            Nav.NavigateTo($"/events/{EventId}/already-registered");
            return;
        }

        AttendanceTracker.Register(EventId, Registration.Email);
        SessionStore.SaveUser(Registration);

        Nav.NavigateTo($"/events/{EventId}/confirmation");
    }
}
